 151:   /**
 152:    * Connect to an SMTP server
 153:    *
 154:    * SMTP CODE SUCCESS: 220
 155:    * SMTP CODE FAILURE: 421
 156:    * @access public
 157:    * @param string $host SMTP server IP or host name
 158:    * @param int $port The port number to connect to, or use the default port if not specified
 159:    * @param int $timeout How long to wait for the connection to open
 160:    * @param array $options An array of options compatible with stream_context_create()
 161:    * @return bool
 162:    */
 163:   public function Connect($host, $port = 0, $timeout = 30, $options = array()) {
 164:     // Clear errors to avoid confusion
 165:     $this->error = null;
 166: 
 167:     // Make sure we are __not__ connected
 168:     if($this->connected()) {
 169:       // Already connected, generate error
 170:       $this->error = array('error' => 'Already connected to a server');
 171:       return false;
 172:     }
 173: 
 174:     if(empty($port)) {
 175:       $port = $this->SMTP_PORT;
 176:     }
 177: 
 178:     // Connect to the SMTP server
 179:     $errno = 0;
 180:     $errstr = '';
 181:     $socket_context = stream_context_create($options);
 182:     //Need to suppress errors here as connection failures can be handled at a higher level
 183:     $this->smtp_conn = @stream_socket_client($host.":".$port, $errno, $errstr, $timeout, STREAM_CLIENT_CONNECT, $socket_context);
 184: 
 185:     // Verify we connected properly
 186:     if(empty($this->smtp_conn)) {
 187:       $this->error = array('error' => 'Failed to connect to server',
 188:                            'errno' => $errno,
 189:                            'errstr' => $errstr);
 190:       if($this->do_debug >= 1) {
 191:         $this->edebug('SMTP -> ERROR: ' . $this->error['error'] . ": $errstr ($errno)");
 192:       }
 193:       return false;
 194:     }
 195: 
 196:     // SMTP server can take longer to respond, give longer timeout for first read
 197:     // Windows does not have support for this timeout function
 198:     if(substr(PHP_OS, 0, 3) != 'WIN') {
 199:       $max = ini_get('max_execution_time');
 200:       if ($max != 0 && $timeout > $max) { // Don't bother if unlimited
 201:         @set_time_limit($timeout);
 202:       }
 203:       stream_set_timeout($this->smtp_conn, $timeout, 0);
 204:     }
 205: 
 206:     // get any announcement
 207:     $announce = $this->get_lines();
 208: 
 209:     if($this->do_debug >= 2) {
 210:       $this->edebug('SMTP -> FROM SERVER:' . $announce);
 211:     }
 212: 
 213:     return true;
 214:   }

 466:   /**
 467:    * Works like hash_hmac('md5', $data, $key) in case that function is not available
 468:    * @access protected
 469:    * @param string $data
 470:    * @param string $key
 471:    * @return string
 472:    */
 473:   protected function hmac($data, $key) {
 474:       if (function_exists('hash_hmac')) {
 475:           return hash_hmac('md5', $data, $key);
 476:       }
 477: 
 478:       // The following borrowed from http://php.net/manual/en/function.mhash.php#27225
 479: 
 480:       // RFC 2104 HMAC implementation for php.
 481:       // Creates an md5 HMAC.
 482:       // Eliminates the need to install mhash to compute a HMAC
 483:       // Hacked by Lance Rushing
 484: 
 485:       $b = 64; // byte length for md5
 486:       if (strlen($key) > $b) {
 487:           $key = pack('H*', md5($key));
 488:       }
 489:       $key  = str_pad($key, $b, chr(0x00));
 490:       $ipad = str_pad('', $b, chr(0x36));
 491:       $opad = str_pad('', $b, chr(0x5c));
 492:       $k_ipad = $key ^ $ipad ;
 493:       $k_opad = $key ^ $opad;
 494: 
 495:       return md5($k_opad  . pack('H*', md5($k_ipad . $data)));
 496:   }

 540:   /**
 541:    * Issues a data command and sends the msg_data to the server
 542:    * finializing the mail transaction. $msg_data is the message
 543:    * that is to be send with the headers. Each header needs to be
 544:    * on a single line followed by a <CRLF> with the message headers
 545:    * and the message body being separated by and additional <CRLF>.
 546:    *
 547:    * Implements rfc 821: DATA <CRLF>
 548:    *
 549:    * SMTP CODE INTERMEDIATE: 354
 550:    *     [data]
 551:    *     <CRLF>.<CRLF>
 552:    *     SMTP CODE SUCCESS: 250
 553:    *     SMTP CODE FAILURE: 552, 554, 451, 452
 554:    * SMTP CODE FAILURE: 451, 554
 555:    * SMTP CODE ERROR  : 500, 501, 503, 421
 556:    * @access public
 557:    * @param string $msg_data
 558:    * @return bool
 559:    */
 560:   public function Data($msg_data) {
 561:     $this->error = null; // so no confusion is caused
 562: 
 563:     if(!$this->connected()) {
 564:       $this->error = array(
 565:               'error' => 'Called Data() without being connected');
 566:       return false;
 567:     }
 568: 
 569:     $this->client_send('DATA' . $this->CRLF);
 570: 
 571:     $rply = $this->get_lines();
 572:     $code = substr($rply, 0, 3);
 573: 
 574:     if($this->do_debug >= 2) {
 575:       $this->edebug('SMTP -> FROM SERVER:' . $rply);
 576:     }
 577: 
 578:     if($code != 354) {
 579:       $this->error =
 580:         array('error' => 'DATA command not accepted from server',
 581:               'smtp_code' => $code,
 582:               'smtp_msg' => substr($rply, 4));
 583:       if($this->do_debug >= 1) {
 584:         $this->edebug('SMTP -> ERROR: ' . $this->error['error'] . ': ' . $rply);
 585:       }
 586:       return false;
 587:     }
 588: 
 589:     /* the server is ready to accept data!
 590:      * according to rfc 821 we should not send more than 1000
 591:      * including the CRLF
 592:      * characters on a single line so we will break the data up
 593:      * into lines by \r and/or \n then if needed we will break
 594:      * each of those into smaller lines to fit within the limit.
 595:      * in addition we will be looking for lines that start with
 596:      * a period '.' and append and additional period '.' to that
 597:      * line. NOTE: this does not count towards limit.
 598:      */
 599:

1042:    * With SMTP we can tell if we have more lines to read if the
