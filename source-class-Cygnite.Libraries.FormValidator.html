 93:         foreach($this->mandatories as $key=>$val)

 95:             if(!array_key_exists($val,$items))
 96:             {
 97:                 $havefailures = true;

 99:             }
100:         }

103:         foreach($this->equal as $key=>$val)
104:         {
105:             //check that the equals field exists
106:             if(!array_key_exists($key,$items))
107:             {
108:                 $havefailures = true;
109:                 $this->addError($val);
110:             }

112:             //check that the field it's supposed to equal exists
113:             if(!array_key_exists($val,$items))
114:             {
115:                 $havefailures = true;
116:                 $this->addError($val);
117:             }
118: 
119:             //Check that the two fields are equal
120:             if($items[$key] != $items[$val])
121:             {
122:                 $havefailures = true;
123:                 $this->addError($key);
124:             }
125:         }
126: 
127:         foreach($this->validations as $key=>$val)
128:         {
129:                 //An empty value or one that is not in the list of validations or one that is not in our list of mandatories
